```

                                              │
Step 9: token estimation (M-04) ──▶ Step 10: 테스트 가능성 개선 (L-01+)
```

```python
# AgentStateGraph 클래스에 새 메서드 (L380 이후)

def _restore_checkpoint(self, ctx: StateContext) -> Optional[str]:
    """직전 체크포인트에서 상태 복원.
    
    Returns:
        복원된 상태명. 실패하면 None.
    """
    if not self._checkpoint_db_path:
        return None
    
    conn = sqlite3.connect(self._checkpoint_db_path)
    try:
        cursor = conn.execute(
            "SELECT state_name, checkpoint_data FROM checkpoints "
            "WHERE run_id = ? ORDER BY checkpoint_seq DESC LIMIT 1",
            (ctx.run_id,)
        )
        row = cursor.fetchone()
        if row:
            state_name, data = row
            # 체크포인트 데이터에서 상태 복원
            return state_name
        return None
    except sqlite3.Error as e:
        logger.warning(f"Checkpoint restore failed: {e}")
        return None
    finally:
        conn.close()
```

```python
# engine/tool_loader.py

class ToolLoader:
    """플러그인 기반 도구 로딩. 각 도구 모듈이 독립적으로 discovery됨."""
    
    def __init__(self, tool_registry: ToolRegistry, project_root: str):
        self.registry = tool_registry
        self.project_root = project_root
        self._errors: List[str] = []  # 개별 도구 로딩 에러 기록
    
    def load_plugins(self, plugin_dir: str) -> Dict[str, bool]:
        """plugin_dir의 auto_skill_* 또는 tool_* 파일을 동적 로딩.
        개별 도구의 import 실패가 다른 도구에 영향 안 미치게."""
        results = {}
        for pyfile in Path(plugin_dir).glob("*.py"):
            try:
                # 각 도구 독립 로딩
                module = self._load_single_tool(pyfile)
                if module:
                    results[str(pyfile)] = True
                else:
                    results[str(pyfile)] = False
            except Exception as e:
                self._errors.append(f"{pyfile.name}: {e}")
                results[str(pyfile)] = False
        return results
    
    def _load_single_tool(self, path: Path) -> Optional[Any]:
        """개별 도구 로딩 — 실패 시 None 반환 (다른 도구 영향 없음)."""
        try:
            import importlib.util
            spec = importlib.util.spec_from_file_location(path.stem, path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            # BaseTool subclass 찾기
            for name in dir(module):
                cls = getattr(module, name)
                if isinstance(cls, type) and issubclass(cls, BaseTool):
                    self.registry.register(cls)
                    return cls
        except Exception as e:
            logger.warning(f"Tool load failed {path.name}: {e}")
        return None
```

```python
def _index_node_recursive(self, node, parent_id: str, depth: int, tree_id: str):
    """재귀적 AST 노드 인덱싱. depth 제한 없음."""
    if depth > 50:  # 안전장치: 최대 50 depth
        return
    
    node_id = hashlib.sha256(f"{tree_id}:{id(node)}:{depth}".encode()).hexdigest()[:16]
    
    source = getattr(node, 'value', '') if hasattr(node, 'value') else str(getattr(node, 'name', ''))
    content = ast.get_source_segment(self._source, node) or ''
    
    self._db.insert_node(
        node_id=node_id,
        tree_id=tree_id,
        parent_id=parent_id,
        node_type=type(node).__name__,
        title=str(getattr(node, 'name', f'{type(node).__name__}')),
        content=content[:1000],
        depth=depth
    )
    
    # 하위 노드 재귀
    for child in ast.iter_child_nodes(node):
        self._index_node_recursive(child, node_id, depth + 1, tree_id)
```

## Step 9: token estimation 개선

**목표:** len(text)//4 → CJK-aware estimation

### 수정 대상: engine/context_compressor.py

#### 9a. 토큰 추정 개선

```python
def estimate_tokens(self, text: str) -> int:
    """CJK 문자를 고려한 토큰 추 estimate.
    영어: 1 word ≈ 1.3 token
    CJK: 1 char ≈ 1.5 token
    mixed: 가중 평균."""
    
    if not text:
        return 0
    
    # CJK 영역 비율 계산
    cjk_chars = sum(1 for c in text if '\u3040' <= c <= '\u30FF' or  # Hiragana/Katakana
                    '\u4E00' <= c <= '\u9FFF' or  # CJK Unified
                    '\uAC00' <= c <= '\uD7AF')     # Hangul
    
    cjk_ratio = cjk_chars / len(text) if text else 0
    
    # 영어 단어 수 + CJK 문자 수 기반 추정
    words = len(text.split())
    tokens = words * 1.3 * (1 - cjk_ratio) + cjk_chars * 1.5 * cjk_ratio
    
    return max(1, int(tokens))
```
